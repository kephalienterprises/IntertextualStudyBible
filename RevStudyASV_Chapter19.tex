%CHAPTER 19

\chapter{Las Bodas del Cordero}
\lettrine[lines=3,lraise=0.4]{\textcolor{red}{D}}{espués} de estas cosas oí una gran voz de gran compañía%
	\cbiblefoot{Daniel}{10:5-6}{Alzando mis ojos miré, y he aquí un varón\ldots y la voz de sus palabras como la voz de ejército.}
 en el cielo, que decía:
\zz \begin{verse}
Aleluya:\\
Salvación y honra y gloria y potencia al Señor Dios nuestro\\*\vin
\vleftofline{\vnum{2}}porque sus juicios son verdaderos y justos;\\
porque él ha juzgado a la grande ramera,\\*\vin
que ha corrompido la tierra con su fornicación,\\
y ha vengado la sangre de sus siervos de la mano de ella.%
	\cbiblefoottrio{Deuteronomy}{32:43}{Alabad, gentes, a su pueblo, porque él vengará la sangre de sus siervos, Y volverá la venganza a sus enemigos, y expiará su tierra, a su pueblo.}%
				{IIKings}{9:7}{Herirás la casa de Achâb tu señor, para que yo vengue la sangre de mis siervos los profetas, y la sangre de todos los siervos de Jehová, de la mano de Jezabel.}%
				{Psalms}{79:10}{Porque dirán las gentes: ¿Dónde está su Dios? Sea notoria en las gentes, delante de nuestros ojos, la venganza de la sangre de tus siervos, que fue derramada.}
\end{verse}

\noindent
\vnum{3}Y otra vez dijeron:
\begin{verse}
Aleluya.\\
Y su humo subió para siempre jamás.%
	\cbiblefoot{Isaiah}{34:9-10}{Sus arroyos se tornarán en pez [\textit{RVR1960: brea}], y su polvo en azufre, y su tierra en pez ardiente. No se apagará de noche ni de día, perpetuamente subirá su humo: de generación en generación será asolada, nunca jamás pasará nadie por ella.}
\end{verse}
\vnum{4}Y los veinticuatro ancianos y los cuatro animales se postraron en tierra, y adoraron a Dios que estaba sentado sobre el trono, diciendo: 
\begin{verse}
Amén: Aleluya.
\end{verse}
\vnum{5}Y salió una voz del trono, que decía:
\begin{verse}
Load a nuestro Dios\\*\vin
todos sus siervos,\\*
y los que le teméis,\\*\vin
así pequeños como grandes.%
	\footnote{\cbibleref{Psalms}{115:13}{Bendecirá a los que temen a Jehová; a chicos y a grandes.}}
\end{verse}
\vnum{6}Y oí como la voz de una grande compañía, y como el ruido de muchas aguas, y como la voz de grandes truenos, que decía:
\begin{verse}
Aleluya:\\
porque reinó el Señor nuestro Dios Todopoderoso.%
	\footnote{\cbibleref{Psalms}{93:1}{Jehová reina, vistióse de magnificencia, Vistióse Jehová, ciñose de fortaleza; Afirmó también el mundo, que no se moverá.} %
			 \cbiblechvs{Psalms}{99:1}{Jehová reinó, temblarán los pueblos: El está sentado sobre los querubines, conmoveráse la tierra.} %
			 \cbibleref{IChronicles}{16:31}{Alégrense los cielos, y gócese la tierra, y digan en las naciones: Reina Jehová.} %
			 \cbibleref{Zechariah}{14:9}{Jehová será rey sobre toda la tierra. En aquel día Jehová será uno, y uno su nombre.}}\\
\vleftofline{\vnum{7}}Gocémonos y alegrémonos%
	\cbiblefoot{Psalms}{118:24}{Este es el día que hizo Jehová Nos gozaremos y alegraremos en él.}
 y démosle gloria;\\
porque son venidas las bodas del Cordero, y su esposa se ha aparejado.\\
\vleftofline{\vnum{8}}Y le fue dado que se vista de lino fino, limpio y brillante:
\end{verse}
porque el lino fino son las justificaciones de los santos.%
	\cbiblefoot{Isaiah}{61:10}{En gran manera me gozaré en Jehová, mi alma se alegrará en mi Dios; porque me vistió de vestidos de salud, rodeóme de manto de justicia, como a novio me atavió, y como a novia compuesta de sus joyas.}

\vnum{9}Y él me dice: Escribe: Bienaventurados los que son llamados a la cena del Cordero. Y me dijo: Estas palabras de Dios son verdaderas.
\vnum{10}Y yo me eché a sus pies para adorarle. Y él me dijo: Mira que no lo hagas: yo soy siervo contigo, y con tus hermanos que tienen el testimonio de Jesús: adora a Dios; porque el testimonio de Jesús es el espíritu de la profecía.

\vnum{11}Y vi el cielo abierto;%
	\cbiblefoot{Ezekiel}{1:1}{Y fue que a los treinta años\ldots los cielos se abrieron, y vi visiones de Dios.}
 y he aquí un caballo blanco, y el que estaba sentado sobre él, era llamado Fiel y Verdadero, el cual con justicia juzga%
	\footnote{\cbibleref{Psalms}{72:2}{El juzgará tu pueblo con justicia, Y tus afligidos con juicio.} %
			  \cbiblechvs{Psalms}{96:13}{Delante de Jehová que vino: Porque vino a juzgar la tierra. Juzgará al mundo con justicia, Y a los pueblos con su verdad.} %
			  \cbiblechvs{Psalms}{98:9}{Delante de Jehová; porque vino a juzgar la tierra: Juzgará al mundo con justicia, Y a los pueblos con rectitud.} %
			  \cbibleref{Isaiah}{11:4-5}{Sino que juzgará con justicia a los pobres, y argüirá con equidad por los mansos de la tierra: y herirá la tierra con la vara de su boca, y con el espíritu de sus labios matará al impío. Y será la justicia cinto de sus lomos, y la fidelidad ceñidor de sus riñones.}}
 y pelea.
\vnum{12}Y sus ojos eran como llama de fuego,%
	\cbiblefoot{Daniel}{10:6}{Su rostro parecía un relámpago, y sus ojos como antorchas de fuego, y sus brazos y sus pies como de color de metal resplandeciente\ldots}
 y había en su cabeza muchas diademas; y tenía un nombre escrito que ninguno entendía sino él mismo.%
	\cbiblefoot{Isaiah}{62:2-3}{Verán las gentes tu justicia, y todos los reyes tu gloria; y te será puesto un nombre nuevo, que la boca de Jehová nombrará. Y serás corona de gloria en la mano de Jehová, y diadema de reino en la mano del Dios tuyo.}
\vnum{13}Y estaba vestido de una ropa teñida en sangre:%
	\cbiblefoot{Isaiah}{63:1-6}{¿Quién es éste que viene de Edom, de Bosra con vestidos bermejos? ¿éste hermoso en su vestido, que marcha en la grandeza de su poder? Yo, el que hablo en justicia, grande para salvar. ¿Por qué es bermejo tu vestido, y tus ropas como del que ha pisado en lagar? Pisado he yo solo el lagar, y de los pueblos nadie fue conmigo: pisélos con mi ira, y hollélos con mi furor; y su sangre salpicó mis vestidos, y ensucié todas mis ropas. Porque el día de la venganza está en mi corazón, y el año de mis redimidos es venido. Y miré y no había quien ayudará, y maravilléme que no hubiera quien sustentase: y salvóme mi brazo, y sostúvome mi ira. Y con mi ira hollé los pueblos, y embriaguélos de mi furor, y derribé a tierra su fortaleza.\label{fn_is63}}
 y su nombre es llamado \textsc{el verbo de dios}.
\vnum{14}Y los ejércitos que están en el cielo le seguían en caballos blancos, vestidos de lino finísimo, blanco y limpio.
\vnum{15}Y de su boca sale una espada aguda,%
	\cbiblefoot{Isaiah}{49:2}{Puso mi boca como espada aguda, cubrióme con la sombra de su mano; y púsome por saeta limpia, guardóme en su aljaba.}
 para herir con ella las gentes: y él los regirá con vara de hierro;%
	\cbiblefoot{Psalms}{2:9}{Quebrantarlos has con vara de hierro: como vaso de alfarero los desmenuzarás.}
 y él pisa el lagar del vino del furor, y de la ira del Dios%
	\footnote{cf. \cbiblerefonly{Isaiah}{63:1-6} citado en \footref{fn_is63}}
 Todopoderoso. 
\vnum{16}Y en su vestidura y en su muslo tiene escrito este nombre: \textsc{rey de reyes y señor de señores}.%
	\cbiblefoot{Deuteronomy}{10:17}{Porque Jehová vuestro Dios es Dios de dioses, y Señor de señores, Dios grande, poderoso, y terrible, que no acepta persona, ni toma cohecho.}

\vnum{17}Y vi un ángel que estaba en el sol, y clamó con gran voz, diciendo a todas las aves que volaban por medio del cielo:%
	\cbiblefootduosbch{Ezekiel}{39:4}{Sobre los montes de Israel caerás tú, y todas tus compañías, y los pueblos que fueron contigo: a toda ave y a toda cosa que vuela, y a las bestias del campo, te he dado por comida.}%
							{39:17-20}{Tú, hijo del hombre, así ha dicho el Señor Jehová: Di a las aves, a todo volátil, y a toda bestia del campo: Juntaos, y venid; reuníos de todas partes a mí víctima que os sacrifico, un sacrificio grande sobre los montes de Israel, y comeréis carne y beberéis sangre. Comeréis carne de fuertes, y beberéis sangre de príncipes de la tierra; de carneros, de corderos, de machos de cabrío, de bueyes, de toros, engordados todos en Basán. Y comeréis gordura hasta hartaros y beberéis hasta embriagaros sangre, de mi sacrificio que yo os sacrifiqué. Y os hartaréis sobre mi mesa, de caballos, y de caballeros fuertes, y de todos hombres de guerra, dice el Señor Jehová.}
 Venid, y congregaos a la cena del gran Dios, 
\vnum{18}para que comáis carnes de reyes, y de capitanes, y carnes de fuertes, y carnes de caballos, y de los que están sentados sobre ellos; y carnes de todos, libres y siervos, de pequeños y de grandes.

\vnum{19}Y vi la bestia, y los reyes de la tierra y sus ejércitos, congregados para hacer guerra contra el que estaba sentado sobre el caballo, y contra su ejército.%
	\cbiblefoot{Psalms}{2:2}{Estarán los reyes de la tierra, Y príncipes consultarán unidos Contra Jehová, y contra su ungido\ldots}
\vnum{20}Y la bestia fue presa,%
	\cbiblefoot{Daniel}{7:11-12}{Yo entonces miraba a causa de la voz de las grandes palabras que hablaba el cuerno; miraba hasta tanto que mataron la bestia, y su cuerpo fue deshecho, y entregado para ser quemado en el fuego. Habían también quitado a las otras bestias su señorío, y les había sido dada prolongación de vida hasta cierto tiempo.}
 y con ella el falso profeta que había hecho las señales delante de ella, con las cuales había engañado a los que tomaron la señal de la bestia, y habían adorado su imagen. Estos dos fueron lanzados vivos%
	\cbiblefootduo{Numbers}{16:33}{Ellos, con todo lo que tenían, descendieron vivos al abismo, y cubriólos la tierra, y perecieron de en medio de la congregación.}%
				  {Psalms}{55:15}{Condenados sean a muerte, Desciendan vivos al infierno: Porque maldades hay en su compañía, entre ellos.}
 dentro de un lago de fuego ardiendo en azufre.%
	\cbiblefoot{Isaiah}{30:33}{Porque Topheth ya de tiempo está diputada y aparejada para el rey, profunda y ancha; cuyo foco es de fuego, y mucha leña; el soplo de Jehová, como torrente de azufre, la enciende.}
\vnum{21}Y los otros fueron muertos con la espada que salía de la boca%
	\cbiblefoot{Isaiah}{11:4}{Sino que juzgará con justicia a los pobres, y argüirá con equidad por los mansos de la tierra: y herirá la tierra con la vara de su boca, y con el espíritu de sus labios matará al impío.}
 del que estaba sentado sobre el caballo, y todas las aves fueron hartas de las carnes de ellos.
