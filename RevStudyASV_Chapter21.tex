%CHAPTER 21

\chapter{Un Cielo Nuevo y \mbox{una Tierra Nueva}}
\lettrine[lines=3,slope=-0.5em,loversize=0.1]{\textcolor{red}{Y}}{\hspace{0.5em} vi} un cielo nuevo, y una tierra nueva:%
	\cbiblefootduosb{Isaiah}{65:17}{Porque he aquí que yo crío nuevos cielos y nueva tierra: y de lo primero no habrá memoria, ni más vendrá al pensamiento.}%
				{66:22}{Porque como los cielos nuevos y la nueva tierra, que yo hago, permanecen delante de mí, dice Jehová, así permanecerá vuestra simiente y vuestro nombre.} %
 porque el primer cielo y la primera tierra se fueron, y el mar ya no es. 
\vnum{2}Y yo Juan vi la santa ciudad,%
	\cbiblefootduosb{Nehemiah}{11:1}{Habitaron los príncipes del pueblo en Jerusalem; mas el resto del pueblo echó suertes para traer uno de diez que morase en Jerusalem, ciudad santa, y las nueve partes en las otras ciudades.}%
					{11:18}{Todos los Levitas en la santa ciudad\ldots} %
 Jerusalem nueva, que descendía del cielo, de Dios, dispuesta como una esposa ataviada para su marido.%
	\cbiblefoot{Isaiah}{61:10}{En gran manera me gozaré en Jehová, mi alma se alegrará en mi Dios; porque me vistió de vestidos de salud, rodeóme de manto de justicia, como a novio me atavió, y como a novia compuesta de sus joyas.}
\vnum{3}Y oí una gran voz del cielo que decía:
\begin{verse}
He aquí el tabernáculo de Dios con los hombres,\\*\vin
y morará con ellos;%
	\footnote{
		\cbibleref{Zechariah}{2:10}{Canta y alégrate, hija de Sión: porque he aquí vengo, y moraré en medio de ti, ha dicho Jehová.} %
		\cbibleref{Leviticus}{26:12}{Andaré entre vosotros, y yo seré vuestro Dios, y vosotros seréis mi pueblo.} %
		\cbibleref{Ezekiel}{37:27}{Estará en ellos mi tabernáculo, y seré a ellos por Dios, y ellos me serán por pueblo.}%
	}\\*\vin
y ellos serán su pueblo,%
	\footnote{
		\cbibleref{Jeremiah}{31:1}{En aquel tiempo, dice Jehová, yo seré por Dios a todos los linajes de Israel, y ellos me serán a mí por pueblo.} %
		\cbibleref{Ezekiel}{11:20}{Para que anden en mis ordenanzas, y guarden mis juicios y los cumplan, y me sean por pueblo, y yo sea a ellos por Dios.} %
		\cbibleref{Zechariah}{8:8}{Traerélos, y habitarán en medio de Jerusalem; y me serán por pueblo, y yo seré a ellos por Dios con verdad y con justicia.} %
		\cbibleref{Psalms}{95:7}{Porque él es nuestro Dios; Nosotros el pueblo de su dehesa, y ovejas de su mano.} %		
	}\\*\vin
y el mismo Dios será su Dios con ellos.\\
\vleftofline{\vnum{4}}Y limpiará Dios toda lágrima de los ojos de ellos;%
	\cbiblefootduo{Isaiah}{25:8}{Destruirá a la muerte para siempre; y enjugará el Señor toda lágrima de todos los rostros: y quitará la afrenta de su pueblo de toda la tierra: porque Jehová lo ha dicho.}%
	{Jeremiah}{31:16}{Así ha dicho Jehová: Reprime tu voz del llanto, y tus ojos de las lágrimas; porque salario hay para tu obra, dice Jehová, y volverán de la tierra del enemigo.}\\*\vin
y la muerte no será más;\\*\vin
y no habrá más llanto, ni clamor, ni dolor:%
	\footnote{\cbibleref{Isaiah}{35:10}{Los redimidos de Jehová volverán, y vendrán a Sión con alegría; y gozo perpetuo será sobre sus cabezas: y retendrán el gozo y alegría, y huirá la tristeza y el gemido.} %
		      \cbiblechvs{Isaiah}{65:19}{Alegraréme con Jerusalem, y gozaréme con mi pueblo; y nunca más se oirán en ella voz de lloro, ni voz de clamor.}}\\*\vin
porque las primeras cosas son pasadas.%
	\footnote{\cbibleref{Isaiah}{43:18}{No os acordéis de las cosas pasadas, ni traigáis a memoria las cosas antiguas.} %
		      cf. \cbiblerefonly{Isaiah}{65:17} citado arriba}
\end{verse}

\vnum{5}Y el que estaba sentado en el trono dijo: He aquí, yo hago nuevas todas las cosas.%
	\cbiblefoot{Isaiah}{43:19}{He aquí que yo hago cosa nueva: presto saldrá a luz: ¿no la sabréis? Otra vez pondré camino en el desierto, y ríos en la soledad.} %
 Y me dijo: Escribe; porque estas palabras son fieles y verdaderas. 
\vnum{6}Y díjome: Hecho es. Yo soy Alpha y Omega, el principio y el fin. Al que tuviere sed, yo le daré de la fuente del agua de vida gratuitamente.%
	\footnote{\cbibleref{Isaiah}{55:1}{A todos los sedientos: Venid a las aguas; y los que no tienen dinero, venid, comprad, y comed. Venid, comprad, sin dinero y sin precio, vino y leche.} %
			  \cbibleref{Jeremiah}{2:13}{ Porque dos males ha hecho mi pueblo: dejáronme a mí, fuente de agua viva, por cavar para sí cisternas, cisternas rotas que no detienen aguas.} %
			  \cbibleref{Zechariah}{14:8}{Acontecerá también en aquel día, que saldrán de Jerusalem aguas vivas; la mitad de ellas hacia la mar oriental, y la otra mitad hacia la mar occidental, en verano y en invierno.}} %
\vnum{7}El que venciere, poseerá todas las cosas; y yo seré su Dios, y él será mi hijo.%
	\cbiblefoot{IISamuel}{7:14}{Yo le seré a él padre, y él me será a mí hijo. Y si él hiciere mal, yo le castigaré con vara de hombres, y con azotes de hijos de hombres;} %
\vnum{8}Mas a los temerosos é incrédulos, a los abominables y homicidas, a los fornicarios y hechiceros, y a los idólatras, y a todos los mentirosos, su parte será en el lago ardiendo con fuego y azufre, que es la muerte segunda.

\vnum{9}Y vino a mí uno de los siete ángeles que tenían las siete copas llenas de las siete postreras plagas, y habló conmigo, diciendo: Ven acá, yo te mostraré la esposa, mujer del Cordero. 
\vnum{10}Y llevóme en Espíritu a un grande y alto monte,%
	\cbiblefoot{Ezekiel}{40:1-2}{En el año veinticinco de nuestro cautiverio\ldots a los catorce años después que la ciudad fué herida, en aquel mismo día fué sobre mí la mano de Jehová, y llevóme allá. En visiones de Dios me llevó a la tierra de Israel, y púsome sobre un monte muy alto, sobre el cual había como edificio de una ciudad al mediodía.} %
 y me mostró la grande ciudad santa de Jerusalem, que descendía del cielo de Dios, 
\vnum{11}teniendo la claridad%
	\footnote{RVR 1960: teniendo la \textit{gloria} de Dios\ldots}
 de Dios:%
	\footnote{\cbibleref{Isaiah}{58:8}{Entonces nacerá tu luz como el alba, y tu salud se dejará ver presto; é irá tu justicia delante de ti, y la gloria de Jehová será tu retaguardia.} %
				\cbiblechvs{Isaiah}{60:1-2}{Levántate, resplandece; que ha venido tu lumbre, y la gloria de Jehová ha nacido sobre ti. Porque he aquí que tinieblas cubrirán la tierra, y oscuridad los pueblos: mas sobre ti nacerá Jehová, y sobre ti será vista su gloria.} %
				\cbibleref{Ezekiel}{43:2-5}{He aquí la gloria del Dios de Israel, que venía de hacia el oriente; y su sonido era como el sonido de muchas aguas, y la tierra resplandecía a causa de su gloria. Y la visión que vi era como la visión, como aquella visión que vi cuando vine para destruir la ciudad: y las visiones eran como la visión que vi junto al río de Chebar; y caí sobre mi rostro. Y la gloria de Jehová entró en la casa por la vía de la puerta que daba cara al oriente. Y alzóme el espíritu, y metióme en el atrio de adentro; y he aquí que la gloria de Jehová hinchió la casa.}} %
 y su luz era semejante a una piedra preciosísima, como piedra de jaspe, resplandeciente como cristal. 
\vnum{12}Y tenía un muro grande y alto con doce puertas;%
	\cbiblefoot{Ezekiel}{48:30-34}{Y estas son las salidas de la ciudad\ldots Y las puertas de la ciudad serán según los nombres de las tribus de Israel: tres puertas al norte: la puerta de Rubén, una; la puerta de Judá, otra; la puerta de Leví, otra. Y a la parte del oriente\ldots tres puertas: la puerta de José, una; la puerta de Benjamín, otra; la puerta de Dan, otra. Y a la parte del mediodía\ldots tres puertas: la puerta de Simeón, una; la puerta de Issachâr, otra; la puerta de Zabulón, otra. Y a la parte del occidente\ldots tres puertas: la puerta de Gad, una; la puerta de Aser, otra; la puerta de Nephtalí, otra.} %
 y en las puertas, doce ángeles,%
	\cbiblefoot{Isaiah}{62:6}{Sobre tus muros, oh Jerusalem, he puesto guardas; todo el día y toda la noche no callarán jamás. Los que os acordáis de Jehová, no ceséis.} %
 y nombres escritos, que son los de las doce tribus de los hijos de Israel.%
	\cbiblefoot{Exodus}{28:21}{Serán aquellas piedra según los nombres de los hijos de Israel, doce según sus nombres; como grabaduras de sello cada una con su nombre, vendrán a ser según las doce tribus.} %
\vnum{13}Al oriente tres puertas; al norte tres puertas; al mediodiá tres puertas; al poniente tres puertas.
\vnum{14}Y el muro de la ciudad tenía doce fundamentos, y en ellos los doce nombres de los doce apóstoles del Cordero. 
\vnum{15}Y el que hablaba conmigo, tenía una medida de una caña de oro para medir la ciudad, y sus puertas, y su muro.%
	\cbiblefoot{Ezekiel}{40:3, 5}{He aquí un varón, cuyo aspecto era como aspecto de metal, y tenía un cordel de lino en su mano, y una caña de medir\ldots de seis codos, de a codo y palmo: y midió\ldots} %
\vnum{16}Y la ciudad está situada y puesta en cuadro, y su largura es tanta como su anchura:%
	\cbiblefoot{Ezekiel}{43:16}{El altar tenía doce codos de largo, y doce de ancho, cuadrado a sus cuatro lados.} %
 y él midió la ciudad con la caña, doce mil estadios: la largura y la altura y la anchura de ella son iguales.%
	\cbiblefoot{Ezekiel}{48:16}{Y estas serán sus medidas: a la parte del norte cuatro mil y quinientas cañas, y a la parte del mediodía cuatro mil y quinientas, y a la parte del oriente cuatro mil y quinientas, y a la parte del occidente cuatro mil y quinientas.}
\vnum{17}Y midió su muro, ciento cuarenta y cuatro codos, de medida de hombre, la cual es del ángel. 
\vnum{18}Y el material de su muro era de jaspe: mas la ciudad era de oro puro, semejante al vidrio limpio. 
\vnum{19}Y los fundamentos del muro de la ciudad estaban adornados de toda piedra preciosa.%
	\cbiblefoottrio{Exodus}{28:17-20}{Y lo llenarás de pedrería con cuatro órdenes de piedras: un orden de una piedra sárdica, un topacio, y un carbunclo; será el primer orden; el segundo orden, una esmeralda, un zafiro, y un diamante; el tercer orden, un rubí, un ágata, y una amatista; y el cuarto orden, un berilo, un onix, y un jaspe: estarán engastadas en oro en sus encajes.}%
				{Isaiah}{54:11-12}{Pobrecita, fatigada con tempestad, sin consuelo; he aquí que yo cimentaré tus piedras sobre carbunclo, y sobre zafiros te fundaré. Tus ventanas pondré de piedras preciosas, tus puertas de piedras de carbunclo, y todo tu término de piedras de buen gusto.}%
				{Ezekiel}{28:12-14}{Tú\ldots acabado de hermosura. En Edén, en el huerto de Dios estuviste: toda piedra preciosa fué tu vestidura; el sardio, topacio, diamante, crisólito, onique, y berilo, el zafiro, carbunclo, y esmeralda, y oro, los primores de tus tamboriles y pífanos estuvieron apercibidos para ti en el día de tu creación. Tú, querubín grande, cubridor: y yo te puse; en el santo monte de Dios estuviste; en medio de piedras de fuego has andado.} %
 El primer fundamento era jaspe; el segundo, zafiro; el tercero, calcedonia; el cuarto, esmeralda; 
\vnum{20}El quinto, sardónica; el sexto, sardio; el séptimo, crisólito; el octavo, berilo; el nono, topacio; el décimo, crisopraso; el undécimo, jacinto; el duodécimo, amatista. 
\vnum{21}Y las doce puertas eran doce perlas, en cada una, una; cada puerta era de una perla. Y la plaza de la ciudad era de oro puro como vidrio trasparente. 
\vnum{22}Y no vi en ella templo; porque el Señor Dios Todopoderoso es el templo de ella, y el Cordero. 
\vnum{23}Y la ciudad no tenía necesidad de sol, ni de luna, para que resplandezcan en ella: porque la claridad de Dios la iluminó, y el Cordero era su lumbrera.%
	\footnote{\cbibleref{Isaiah}{60:1}{Levántate, resplandece; que ha venido tu lumbre, y la gloria de Jehová ha nacido sobre ti.} %
	\cbiblevs{Isaiah}{60:19-20}{El sol nunca más te servirá de luz para el día, ni el resplandor de la luna te alumbrará; sino que Jehová te será por luz perpetua, y el Dios tuyo por tu gloria. No se pondrá jamás tu sol, ni menguará tu luna: porque te será Jehová por luz perpetua, y los días de tu luto serán acabados.} %
	\cbiblechvs{Isaiah}{24:23}{La luna se avergonzará, y el sol se confundirá, cuando Jehová de los ejércitos reinare en el monte de Sión, y en Jerusalem, y delante de sus ancianos fuere glorioso.}} %
\vnum{24}Y las naciones que hubieren sido salvas andarán en la lumbre de ella:%
	\cbiblefoot{Isaiah}{60:3}{Andarán las gentes a tu luz, y los reyes al resplandor de tu nacimiento.}
 y los reyes de la tierra traerán su gloria y honor a ella.
	\footnote{\cbibleref{Isaiah}{60:5}{Entonces verás y resplandecerás; y se maravillará y ensanchará tu corazón, que se haya vuelto a ti la multitud de la mar, y la fortaleza de las gentes haya venido a ti.} %
			\cbibleref{Psalms}{68:29-32}{Por razón de tu templo en Jerusalem Los reyes te ofrecerán dones\ldots Vendrán príncipes de Egipto; Etiopía apresurará sus manos a Dios. Reinos de la tierra, cantad a Dios, cantad al Señor (Selah).} %
			\cbibleref{Psalms}{72:10-11}{Los reyes de Tharsis y de las islas traerán presentes: los reyes de Sheba y de Seba ofrecerán dones. Y arrodillarse han a él todos los reyes; le servirán todas las gentes.}} %
\vnum{25}Y sus puertas nunca serán cerradas de día,%
	\cbiblefoot{Isaiah}{60:11}{Tus puertas estarán de continuo abiertas, no se cerrarán de día ni de noche, para que sea traída a ti fortaleza de gentes, y sus reyes conducidos.} %
 porque allí no habrá noche.%	
	\cbiblefoot{Zechariah}{14:7}{Será un día, el cual es conocido de Jehová, que ni será día ni noche; mas acontecerá que al tiempo de la tarde habrá luz.}
\vnum{26}Y llevarán la gloria y la honra de las naciones a ella. 
\vnum{27}No entrará en ella ninguna cosa sucia, ó que hace abominación y mentira;%
	\cbiblefootduosb{Isaiah}{35:8}{Habrá allí calzada y camino, y será llamado Camino de Santidad; no pasará por él inmundo; y habrá para ellos en él quien los acompañe, de tal manera que los insensatos no yerren.}%
				{52:1}{Despierta, despierta, vístete tu fortaleza, oh Sión; vístete tu ropa de hermosura, oh Jerusalem, ciudad santa: porque nunca más acontecerá que venga a ti incircunciso ni inmundo.} %
 sino solamente los que están escritos en el libro de la vida del Cordero.%
	\cbiblefoot{Isaiah}{4:3}{Acontecerá que el que quedare en Sión, y el que fuere dejado en Jerusalem, será llamado santo; todos los que en Jerusalem están escritos entre los vivientes;}
